from manimlib import Scene, Circle, Square, Triangle, Text
from manimlib.constants import RED, BLUE, GREEN, UP

class MyScene(Scene):
    def construct(self):
        # Title that persists
        title = Text("My Animation").to_edge(UP)
        self.add(title)
        
        # Scene 1
        obj1 = Circle(color=RED)
        obj2 = Square(color=BLUE)
        obj3 = Triangle(color=GREEN)
        self.add(obj1, obj2, obj3)
        self.wait(1)
        
        # Transition - keep only title and obj2
        self.clear_all_except(title, obj2)
        self.wait(1)
        
        # Scene 2
        new_obj = Circle(color=GREEN)
        self.add(new_obj)
        self.wait(1)
