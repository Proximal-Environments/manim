USAGE EXAMPLE

from manimlib import Scene
from manimlib.mobject.geometry import Circle, Square

class Example(Scene):
    def construct(self):
        circle = Circle()
        square = Square()
        
        self.add(circle, square)
        self.wait(1)
        
        # Keep only circle
        self.clear_all_except(circle)
        self.wait(1)
